import styled from "styled-components";

const Title = styled.h1`color: tomato`

const Header = () => (
  <Container>
    <Title>Hey</Title>
  </Container>
)

`;

–if you're already using Webpack–

CSS Modules are enabled through Webpack's css-loader

// webpack.config.js
{
  test: /\.css$/,
  use: [
    {
      loader: 'css-loader',
      options: {
        modules: true,
        localIdentName: '[path][name]__[local]--[hash:base64:5]'
      }
    }
  ]
}

// Header.jsx
import 'Footer.scss' // Load as regular CSS
import css from 'Footer.module.scss' // Load as local CSS

// You'll be still able to share your constants in CSS Modules
